<canvas id="terrain" width="500" height="300"></canvas>
<script>
let canvas = document.getElementById("terrain")
let ctx = canvas.getContext("2d")
function generateTerrain(){
for(let x = 0; x < canvas.width; x++){
let height = Math.random() * 150 + 50
ctx.fillStyle = "green"
ctx.fillRect(x, canvas.height - height, 1, height)
}
}
generateTerrain()
</script>
